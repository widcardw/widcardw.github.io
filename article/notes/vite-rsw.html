                                  # Vite Rsw 使用入门

                                  # 来源

                                  https://github.com/rwasm/vite-plugin-rswopen in new window

                                  这个方法要比手动改 vite.config.ts 和 tsconfig.json 方便的多，感谢库作者

                                  # 入门

                                  # 0. Rust 换源

                                  编辑 ~/.cargo/config

                                  $ vi ~/.cargo/config
                                  1

                                  写入以下内容

                                  [source.crates-io]
registry = "https://github.com/rust-lang/crates.io-index"
replace-with = 'ustc'
[source.ustc]
registry = "git://mirrors.ustc.edu.cn/crates.io-index"
[net]
git-fetch-with-cli = true

                                  # 1. 安装依赖

                                  首先需要安装好各种各样的 rust 依赖，此处全部使用 cargo 来安装

                                  $ cargo install wasm-pack
$ cargo install rsw

                                  2

                                  # 2. 初始化一个 Web 项目

                                  新建一个文件夹

                                  $ mkdir vite-rsw-test && cd vite-rsw-test
                                  1

                                  初始化 pnpm

                                  $ pnpm init
                                  1

                                  安装 vite 和 vite-plugin-rsw

                                  $ pnpm i vite
$ pnpm i vite-plugin-rsw
                                  1
                                  2

                                  初始化 git

                                  $ git init
                                  1

                                  # 3. 配置 Web 项目

                                  新建 vite.config.ts，编写官网给的内容

                                  import { defineConfig } from 'vite';
import ViteRsw from 'vite-plugin-rsw';

export default defineConfig({
  plugins: [
    ViteRsw(),
  ],
})

                                  # 4. 初始化 Rust 项目

                                  初始化 rsw.toml

                                  $ rsw init
                                  1

                                  生成 rust crate

                                  # rsw.toml
[new]
# using: wasm-pack | rsw | user, default is `wasm-pack`
using = "wasm-pack"
                                  1
                                  2
                                  3
                                  4
                                  $ rsw new rsw-hello
                                  1

                                  这一步需要在 git init 之后才能成功执行，因为 git fetch 需要当前为一个 git 仓库

                                  编辑 rsw.toml

                                  # link type: npm | yarn | pnpm, default is `npm`
cli = "pnpm"

[[crates]]
name = "rsw-hello"
# <npm|yarn|pnpm> link
# ⚠️ Note: must be set to `true`, default is `false`
link = true

                                  # 6. 启动

                                  启动 Rsw 的监看

                                  $ pnpm run watch
                                  1

                                  启动 vite 前端项目

                                  $ pnpm run dev
                                  1
                                  注意

                                  两个命令行需要分别开启，因为上面一个是监看 Rust 项目的，下面那个是监听 vite 前端项目的

                                  # 7. 补充

                                  对于需要访问内存的操作，根据教程尝试了下面的方法，但是不行。这样做在 DOM 中仍然会报错，说没有 memory 这个模块作为导出。

                                  import { memory } from 'rsw-hello/rsw_hello_bg.wasm'
                                  1

                                  经过翻阅 lencxopen in new window 的源码，发现可以这么做

                                  import init from 'rsw-hello'

const wasmInit = await init()

const memory = wasm.memory
